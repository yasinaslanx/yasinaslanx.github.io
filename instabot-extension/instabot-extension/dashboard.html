💡 Her çalıştırmada 50 kişi unfollow eder, kaldığı yerden devam eder. 1-2 saat bekleyip tekrar çalıştır.

// 50'li unfollow — kaldigi yerden devam eder
(async()=>{
  if(!window.__ig?.nonfollowers?.length){console.error("Once analiz!");return;}
  const LIMIT=50;
  const KEY="ig_uf_"+window.__ig.username;
  const done=JSON.parse(localStorage.getItem(KEY)||"[]");
  const dSet=new Set(done);
  const batch=window.__ig.nonfollowers.filter(u=>!dSet.has(u.username)).slice(0,LIMIT);
  const csrf=document.cookie.match(/csrftoken=([^;]+)/)?.[1]||"";
  console.log("Toplam:",window.__ig.nonfollowers.length,"Yapilan:",done.length,"Bu tur:",batch.length);
  if(!batch.length){console.log("Tamamlandi!");return;}
  let ok=0,fail=0;
  for(const user of batch){
    try{
      const r=await fetch(`/api/v1/friendships/destroy/${user.id}/`,{
        method:"POST",headers:{
          "x-ig-app-id":"936619743392459","x-csrftoken":csrf,
          "x-requested-with":"XMLHttpRequest","Content-Type":"application/x-www-form-urlencoded"
        }
      });
      if(r.ok){ok++;done.push(user.username);localStorage.setItem(KEY,JSON.stringify(done));console.log("OK ["+(ok+fail)+"/"+batch.length+"] @"+user.username);}
      else if(r.status===429){console.warn("Rate limit 60sn...");await new Promise(r=>setTimeout(r,60000));}
      else{fail++;console.warn("HATA @"+user.username);}
    }catch(e){fail++;}
    await new Promise(r=>setTimeout(r,2000+Math.random()*2000));
  }
  const kalan=window.__ig.nonfollowers.length-done.length;
  console.log("Bitti! OK:"+ok+" Hata:"+fail+" Kalan:"+kalan);
  if(kalan>0)console.log(kalan+" kisi kaldi. 1-2 saat bekle.");
  else console.log("Tum liste tamamlandi!");
})();

(function(){
  if(!window.__ig){console.error("Once analiz!");return;}
  const a=document.createElement("a");
  a.href=URL.createObjectURL(new Blob([JSON.stringify({
    tarih:new Date().toISOString(),
    kullanici:window.__ig.username,
    following:window.__ig.following.length,
    takip_etmeyenler:window.__ig.nonfollowers.map(u=>u.username)
  },null,2)],{type:"application/json"}));
  a.download=`ig_${window.__ig.username}.json`;
  a.click();
})();
